319    def record_startup(
320        self,
321        *,
322        meta: dict[str, Any] | None = None,
323    ) -> str:
324        """Capture a deployment snapshot and record a service-restart intent.
325
326        Call this in your application's startup handler so that every
327        (re)start is automatically visible to agents — they can correlate
328        deploys, dependency changes, and config shifts with runtime
329        behaviour changes.
330
331        Returns the snapshot event_id.
332        """
333        snapshot_id = self.snapshot()
334
335        startup_inputs: dict[str, Any] = {
336            "service": self._config.service_name,
337            "snapshot_id": self._current_snapshot_id or "",
338        }
339        if meta:
340            startup_inputs.update(meta)
341
342        with context.trace_context(new_id()):
343            self._record_semantic_event(
344                "service-restart",
345                "intent",
346                {
347                    "semantic": {
348                        "type": "intent",
349                        "intent": "service-restart",
350                        "inputs": startup_inputs,
351                        "expected": {"result": "service_ready"},
352                    }
353                },
354            )
355
356        return snapshot_id

358    def snapshot(self) -> str:
359        """Capture deployment snapshot. Returns snapshot_id."""
360        git_sha = self._run_cmd("git", "rev-parse", "HEAD")
361        git_diff = self._run_cmd("git", "diff", "HEAD")
362
363        deps = self._capture_dependencies()
364        runtime = f"python {sys.version}"
365        env_vars = {
366            k: os.environ.get(k, "")
367            for k in self._config.env_safelist
368            if k in os.environ
369        }
370
371        event = Event.snapshot(
372            git_sha=git_sha,
373            git_diff=git_diff,
374            dependencies=deps,
375            runtime=runtime,
376            env_vars=env_vars,
377            service_name=self._config.service_name,
378        )
379        self._storage.write_event(event)
380        self._current_snapshot_id = event.snapshot_id
381        return event.event_id

489        try:
